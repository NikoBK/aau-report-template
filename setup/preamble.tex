% Set up the documentclass, packages and options

% Has to be set before documentclass to silence warning
\PassOptionsToPackage{pdfpagelabels=true}{hyperref}

% Set up document to use a4 and use the report document class
\documentclass[a4paper]{report}

% Lists (this is where the fun begins)
\usepackage{xstring}
\usepackage{listofitems}


% Apply user settings
\input{setup/user.tex}

% Set up encoding and font
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{lmodern}

% Has to be loaded before csquotes
\usepackage{fvextra}

% Set up language
\usepackage{csquotes}
\if\useenglish0
  \usepackage[danish]{babel}
\else
  \usepackage[english]{babel}
\fi

% Color
\usepackage{xcolor}

% Graphics
\usepackage{graphicx}

% Bibliography
\usepackage[bibencoding=utf8]{biblatex}
\addbibresource{bibliography.bib}

% Refrence to last page
\usepackage{lastpage}

% Math
\usepackage{amsmath}
\usepackage{amssymb}
\DeclareMathSymbol{*}{\mathbin}{symbols}{"01} % Makes the * symbol a times symbol in math mode

% Header
\usepackage{fancyhdr}
\pagestyle{fancy}

% Muticolumns
\usepackage{multicol}

% A box that just works
\usepackage{mdframed}

% Quotation
\usepackage{dirtytalk}

% If then else
\usepackage{ifthen}

% Advanced patching of macros
\usepackage{xpatch}

% Code highlighting
\usepackage{minted}


% Title
\if\useenglish0
  \newcommand{\maintitle}{\titleda}
  \newcommand{\mainsubtitle}{\subtitleda}
\else
  \newcommand{\maintitle}{\titleen}
  \newcommand{\mainsubtitle}{\subtitleen}
\fi

\title{\maintitle\\\large\mainsubtitle}

\input{setup/macro}

\renewcommand{\listingscaption}{\listingname}


% Clickable references and other pdf stuff
% This should be the last package to be included
\usepackage{hyperref}
\hypersetup{%
  plainpages=false,%
  pdfauthor={\authorcomma},%
  pdftitle={\maintitle},%
  pdfsubject={\mainsubtitle},%
  bookmarksnumbered=true,%
  breaklinks,%
  hidelinks,%
  colorlinks=false,%
  citecolor=black,%
  filecolor=black,%
  linkcolor=black,%
  urlcolor=black,%
  pdfstartview=FitH%
}
